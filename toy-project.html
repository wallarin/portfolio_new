		    )
	    );

            "userid" => $this->input->post("userid", true),

            "encrypt" => $this->input->post("encrypt", true),
	    );

$get_data = $this->Admin_Member_model->login_chk($data);

    echo "wrong id";

    echo "성공"; // 응답값

프로젝트명: Nice Tax Free 홈페이지

  • 소속회사: (주) 웨어비즈
  • 주요 업무: 백엔드 및 프론트엔드 담당
  • 담당 역할: 관리자 화면에서 비밀번호 오류 시 계정 잠금 처리 로직 구현
  • 기술 스택: PHP(CodeIgniter), MySQL
  • 업무 기간: 2022년 12월 (약 1주)
  • 개발 인원: 3명

프로젝트 개요:

Nice Tax Free 홈페이지의 관리자 페이지에서 비밀번호 오류 시 계정을 잠그는 기능을 구현하여 보안성을 강화하는 작업을 수행했습니다.

프로젝트 소개 및 주요 기능:

이 프로젝트는 CodeIgniter 프레임워크로 구성된 관리자 페이지에서 보안 강화를 목적으로 비밀번호 오류 시 계정을 잠금 처리하는 로직을 추가하는 것이 주요 목표였습니다. 비밀번호가 일정 횟수 이상 틀릴 경우, 계정이 자동으로 잠기도록 하여 관리자 계정의 보안을 강화했습니다.

기술 스택 및 개발 상세 내용:

  • PHP(CodeIgniter): 백엔드에서 로그인 요청 처리 및 계정 잠금 로직 구현
  • MySQL: DB에서 비밀번호 오류 횟수 및 계정 잠금 여부 관리
  • JavaScript: 프론트엔드에서 로그인 요청 후 백엔드와의 비밀번호 검증을 처리
  • 로그인 시 비밀번호 오류 발생 시, 데이터베이스에서 해당 계정의 비밀번호 오류 횟수를 조회
  • 오류 횟수가 5회를 초과할 경우 계정을 자동으로 잠그는 기능 구현
  • 오류 발생 시마다 실패 횟수를 카운팅하여 데이터베이스에 저장

                        function helloWorld() {
