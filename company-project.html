from apscheduler.schedulers.blocking import BlockingScheduler
from configparser import ConfigParser
import subprocess
import threading
import datetime
import time
from watchdog.observers import Observer
from watchdog.events import FileSystemEventHandler

#config파일 경로
config_path = '/app/config/'
config_name = 'config.ini'

# 글로벌 락 초기화
lock = threading.Lock()


def execute_command(command):
    with lock:  # 락을 사용하여 순차 실행을 보장
        print(f"Executing {command} at {datetime.datetime.now()}")
        subprocess.run(command, shell=True)
        print(f"Finished {command} at {datetime.datetime.now()}")


def load_config_and_schedule(scheduler):
    config = ConfigParser()
    with open(config_path + config_name, encoding='utf-8') as f:
        config.read_file(f)

    scheduler.remove_all_jobs()  # 기존에 스케쥴된 모든 작업을 제거

    # 모든 설정된 스케줄 읽기
    for section in config.sections():
        command = config.get(section, 'command')
        start_time = config.get(section, 'start_time', fallback=None)
        if start_time:
            start_time = datetime.datetime.strptime(start_time, '%Y-%m-%d %H:%M:%S')

        # interval 스케줄 처리
        if config.has_option(section, 'interval_type'):
            interval_types = config.get(section, 'interval_type').split(',')
            intervals = config.get(section, 'interval').split(',')
            interval_dict = {
                interval_type: int(interval)
                for interval_type, interval in zip(interval_types, intervals)
            }
            scheduler.add_job(execute_command, 'interval', [command], start_date=start_time, **interval_dict)

        # cron 스케줄 처리
        elif config.has_option(section, 'cron_type'):
            cron_type = config.get(section, 'cron_type', fallback='daily')
            day = config.get(section, 'day', fallback=None)
            hour = config.getint(section, 'hour', fallback=0)
            minute = config.getint(section, 'minute', fallback=0)
            cron_dict = {
                'hour': hour,
                'minute': minute
            }
            if day:
                cron_dict['day'] = day
            scheduler.add_job(execute_command, 'cron', [command], start_date=start_time, **cron_dict)


class ChangeHandler(FileSystemEventHandler):
    def on_modified(self, event):
        if config_path + config_name in event.src_path:
            print(f"{event.src_path} has been modified. Reloading the configuration...")
            load_config_and_schedule(scheduler)


def start_monitoring():
    observer = Observer()
    event_handler = ChangeHandler()
    observer.schedule(event_handler, config_path, recursive=False)
    observer.start()
    try:
        while True:
            time.sleep(1)
    except KeyboardInterrupt:
        observer.stop()
    observer.join()


def main():
    global scheduler
    scheduler = BlockingScheduler()
    load_config_and_schedule(scheduler)  # 초기 스케줄 설정
    thread = threading.Thread(target=start_monitoring)
    thread.start()  # 설정 파일 감시 시작
    scheduler.start()  # 스케줄러 시작


if __name__ == '__main__':
    main()

    analyzer_search.append("index=kb_law&sentence=" + search_word);
    JSONObject analyzer_result = argosGet(analyzer_search.toString());
    
    query_result.put("analyzer_word", analyzer_result);
    search_word = URLDecoder.decode(search_word, "UTF-8");
    query_result.put("search_word", search_word);
    ibUtil.setCodeMessage("SHEETDATA1","정상적으로 처리되었습니다.");

@RequestMapping(value = "/adownload.do")
public void downloadFile(@RequestParam("file_no") String file_no, @RequestParam("tb") String relation_Table, HttpServletResponse response) throws Exception {

    Map<String, String> map = new HashMap<String, String>();
    
    map.put("file_no", file_no);
    map.put("relation_Table", relation_Table);
    
    List data = lawService.selectFileDir(map);
    
    if (!data.isEmpty()) {
        EgovMap egovMap = (EgovMap) data.get(0);
        String file_Path = propertyService.getString("file.download.rootdir") + egovMap.get("url").toString();
        //String file_Path = "D:\\hello\\20081031_법무법인검토_김장";
        
        File file = new File(file_Path);
        
        if (file.exists()) {
            response.setContentType("application/octet-stream");
            response.setHeader("Content-Disposition", "attachment; filename=\"" + URLEncoder.encode(egovMap.get("name").toString(), "UTF-8") + "\"");
            
            OutputStream outputStream = response.getOutputStream();
            FileInputStream inputStream = new FileInputStream(file);
            
            byte[] buffer = new byte[4096];
            int bytesRead;
            while ((bytesRead = inputStream.read(buffer)) != -1) {
                outputStream.write(buffer, 0, bytesRead);
            }
            
            inputStream.close();
            outputStream.close();
        } else {
            response.sendError(HttpServletResponse.SC_NOT_FOUND, "파일이 존재하지 않습니다.");
        }
    } 
}

/*
* @author 웨어비즈
* @brief 검색엔진을 호출해 JSON 결과를 return 받기 위한 HTTP통신 로직 구현
* */
private JSONObject argosGet(String query) throws Exception {
    
    HttpURLConnection conn = null;
    StringBuilder sb = null;
    
    JSONObject responseJson = null;
    
    try {
        URL url = new URL(query);
        
        LOGGER.debug(query.toString());
        conn = (HttpURLConnection) url.openConnection();
        conn.setConnectTimeout(60000);
        conn.setReadTimeout(60000); 
        conn.setRequestMethod("GET");
        
        int responseCode = conn.getResponseCode();
        if (responseCode == 400 || responseCode == 401 || responseCode == 500) {
            LOGGER.error(responseCode + "Error!");
        } else {
            BufferedReader br = new BufferedReader(new InputStreamReader(conn.getInputStream(), "UTF-8"));
            sb = new StringBuilder();
            String line = "";
            while ((line = br.readLine()) != null) {
                sb.append(line);
            }
            JSONParser parser = new JSONParser();
            responseJson = (JSONObject) parser.parse(sb.toString());
            return responseJson;

        }
    } catch (MalformedURLException e) {
        e.printStackTrace();
    } catch (IOException e) {
        e.printStackTrace();
    } catch (ParseException e) {
        LOGGER.error("not JSON Format response");
        e.printStackTrace();
    } catch (Exception e) {
        e.printStackTrace();
    }
    return responseJson;
}

TEMP_LOG_FILE="$SCHEDULER_DIR/log_temp.log"

check_file_size() {
    local file_path="$1"
    local wait_time=20
    local prev_size=0

    while true; do
        file_size=$(stat -c %s "$file_path")
        if [ "$file_size" -eq "$prev_size" ]; then
            break
        fi
        sleep "$wait_time"
        prev_size="$file_size"
    done
}

cd "$DUMP_DIR/data/bulk"
today=$(date +'%Y-%m-%d')
mv kb_law.txt "kb_law.txt_$today"

find "$DUMP_DIR/data/bulk" -type f -name "kb_law*" -mtime +5 -exec rm {} \;

# DB BULK Process
cd "$DUMP_DIR"
./dump.sh ./conf/bulk/kb_law.conf &> "$TEMP_LOG_FILE" &

check_file_size "$DUMP_DIR/data/bulk/kb_law.txt"

cd "$DUMP_DIR/data/bulk"

# 10MB = 10485760 | 150KB = 153600
if [ -f "kb_law.txt" ] && [ $(stat -c %s "kb_law.txt") -ge 153600 ]; then
    # Elastic Data Delete for Kb_law
    curl -X POST "http://localhost:9200/kb_law/_delete_by_query" -H "Content-Type: application/json" -d '{"query": {"match_all": {}},"_source": {"includes": []}}'

    # Insert Elastic BULK Data
    cd "$INDEXING_DIR"
    ./indexing.sh ./bulk/kb_law.conf
    if [ ! -f $LOG_FILE ]; then
            echo "[`date '+%Y-%m-%d %H:%M:%S'`] kb_law Indexing SUCCESS;" > $LOG_FILE
    else
            echo "[`date '+%Y-%m-%d %H:%M:%S'`] kb_law Indexing SUCCESS;" >> $LOG_FILE
    fi
fi

# Result Record

    echo "[`date '+%Y-%m-%d %H:%M:%S'`] kb_law schedul SUCCESS;" > $LOG_FILE
else
    echo "[`date '+%Y-%m-%d %H:%M:%S'`] kb_law schedul SUCCESS;" >> $LOG_FILE
fi

cd "$SCHEDULER_DIR"
